pragma solidity >=0.4.22 <0.9.0;

contract SimpleStorage {
    uint storedData;

    // Function to set the value
    function set(uint x) public {
        storedData = x;
    }

    // Function to retrieve the value
    function get() public view returns (uint) {
        return storedData;
    }
}
